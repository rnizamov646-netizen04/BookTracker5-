import tkinter as tk
from tkinter import ttk, messagebox
import random
import json
import os
from datetime import datetime

class TaskGeneratorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Random Task Generator")
        self.root.geometry("900x700")
        
        self.predefined_tasks = {
            "Учёба": [
                "Прочитать статью", "Выучить 10 новых слов", "Решить 5 задач по математике",
                "Посмотреть обучающее видео", "Написать конспект", "Подготовиться к экзамену"
            ],
            "Спорт": [
                "Сделать зарядку", "Пробежать 3 км", "Сделать 50 отжиманий",
                "Пойти в спортзал", "Поплавать в бассейне", "Покататься на велосипеде"
            ],
            "Работа": [
                "Составить отчет", "Ответить на письма", "Подготовить презентацию",
                "Созвониться с клиентом", "Обновить базу данных", "Проверить задачи"
            ]
        }
        
        self.history = []
        self.load_history()
        
        if not os.path.exists('data'):
            os.makedirs('data')
        
        self.setup_ui()
    
    def setup_ui(self):
        main_frame = tk.Frame(self.root, padx=20, pady=20)
        main_frame.pack(fill=tk.BOTH, expand=True)
        
        title_label = tk.Label(
            main_frame,
            text="🎯 Random Task Generator",
            font=("Arial", 18, "bold")
        )
        title_label.pack(pady=(0, 20))
        
        generation_frame = tk.LabelFrame(main_frame, text="Генерация задачи", font=("Arial", 10))
        generation_frame.pack(fill=tk.X, pady=(0, 15))
        
        self.task_display = tk.Text(generation_frame, height=4, width=70, font=("Arial", 10))
        self.task_display.pack(padx=10, pady=10)
        self.task_display.config(state=tk.DISABLED)
        
        generate_btn = tk.Button(
            generation_frame,
            text="🎲 Сгенерировать задачу",
            command=self.generate_task,
            font=("Arial", 10),
            bg="#4CAF50",
            fg="white",
            height=2
        )
        generate_btn.pack(pady=(0, 10))
        
        add_frame = tk.LabelFrame(main_frame, text="Добавить свою задачу", font=("Arial", 10))
        add_frame.pack(fill=tk.X, pady=(0, 15))
        
        tk.Label(add_frame, text="Задача:", font=("Arial", 10)).pack(side=tk.LEFT, padx=(10, 5))
        self.task_entry = tk.Entry(add_frame, font=("Arial", 10), width=50)
        self.task_entry.pack(side=tk.LEFT, padx=5)
        
        tk.Label(add_frame, text="Категория:", font=("Arial", 10)).pack(side=tk.LEFT, padx=(20, 5))
        self.category_var = tk.StringVar(value="Учёба")
        category_combo = ttk.Combobox(
            add_frame,
            textvariable=self.category_var,
            values=["Учёба", "Спорт", "Работа"],
            state="readonly",
            width=15
        )
        category_combo.pack(side=tk.LEFT, padx=5)
        
        add_btn = tk.Button(
            add_frame,
            text="➕ Добавить задачу",
            command=self.add_custom_task,
            font=("Arial", 10),
            bg="#2196F3",
            fg="white"
        )
        add_btn.pack(side=tk.LEFT, padx=20, pady=10)
        
        history_frame = tk.LabelFrame(main_frame, text="История задач", font=("Arial", 10))
        history_frame.pack(fill=tk.BOTH, expand=True)
        
        filter_frame = tk.Frame(history_frame)
        filter_frame.pack(fill=tk.X, padx=10, pady=5)
        
        tk.Label(filter_frame, text="Фильтр по категории:", font=("Arial", 10)).pack(side=tk.LEFT)
        self.filter_var = tk.StringVar(value="Все")
        filter_combo = ttk.Combobox(
            filter_frame,
            textvariable=self.filter_var,
            values=["Все", "Учёба", "Спорт", "Работа"],
            state="readonly",
            width=15
        )
        filter_combo.pack(side=tk.LEFT, padx=10)
        filter_combo.bind("<<ComboboxSelected>>", self.filter_history)
        
        buttons_frame = tk.Frame(filter_frame)
        buttons_frame.pack(side=tk.RIGHT)
        
        clear_btn = tk.Button(
            buttons_frame,
            text="🗑️ Очистить историю",
            command=self.clear_history,
            font=("Arial", 9),
            bg="#f44336",
            fg="white"
        )
        clear_btn.pack(side=tk.LEFT, padx=5)
        
        save_btn = tk.Button(
            buttons_frame,
            text="💾 Сохранить историю",
            command=self.save_history,
            font=("Arial", 9),
            bg="#FF9800",
            fg="white"
        )
        save_btn.pack(side=tk.LEFT, padx=5)
        
        load_btn = tk.Button(
            buttons_frame,
            text="📂 Загрузить историю",
            command=self.load_history_ui,
            font=("Arial", 9),
            bg="#9C27B0",
            fg="white"
        )
        load_btn.pack(side=tk.LEFT, padx=5)
        
        columns = ("Дата", "Время", "Задача", "Категория")
        self.history_tree = ttk.Treeview(history_frame, columns=columns, show="headings", height=12)
        
        for col in columns:
            self.history_tree.heading(col, text=col)
            self.history_tree.column(col, width=150)
        
        scrollbar = ttk.Scrollbar(history_frame, orient=tk.VERTICAL, command=self.history_tree.yview)
        self.history_tree.configure(yscrollcommand=scrollbar.set)
        
        self.history_tree.pack(side=tk.LEFT, fill=tk.BOTH, expand=True, padx=10, pady=10)
        scrollbar.pack(side=tk.RIGHT, fill=tk.Y, pady=10)
        
        self.update_history_display()
    
    def generate_task(self):
        category = random.choice(list(self.predefined_tasks.keys()))
        task = random.choice(self.predefined_tasks[category])
        
        self.task_display.config(state=tk.NORMAL)
        self.task_display.delete(1.0, tk.END)
        self.task_display.insert(1.0, f"📋 Задача: {task}\n🏷️ Категория: {category}")
        self.task_display.config(state=tk.DISABLED)
        
        self.add_to_history(task, category)
    
    def add_custom_task(self):
        task = self.task_entry.get().strip()
        category = self.category_var.get()
        
        if not task:
            messagebox.showerror("Ошибка", "Задача не может быть пустой!")
            return
        
        if len(task) > 200:
            messagebox.showerror("Ошибка", "Задача слишком длинная (максимум 200 символов)!")
            return
        
        self.add_to_history(task, category)
        self.task_entry.delete(0, tk.END)
        messagebox.showinfo("Успех", "Задача успешно добавлена!")
    
    def add_to_history(self, task, category):
        now = datetime.now()
        task_data = {
            "date": now.strftime("%Y-%m-%d"),
            "time": now.strftime("%H:%M:%S"),
            "task": task,
            "category": category
        }
        self.history.append(task_data)
        self.update_history_display()
    
    def update_history_display(self):
        for item in self.history_tree.get_children():
            self.history_tree.delete(item)
        
        filter_value = self.filter_var.get()
        filtered_history = self.history if filter_value == "Все" else \
            [task for task in self.history if task["category"] == filter_value]
        
        for task in filtered_history:
            self.history_tree.insert("", tk.END, values=(
                task["date"],
                task["time"],
                task["task"],
                task["category"]
            ))
    
    def filter_history(self, event=None):
        self.update_history_display()
    
    def clear_history(self):
        if messagebox.askyesno("Подтверждение", "Вы уверены, что хотите очистить историю?"):
            self.history = []
            self.update_history_display()
    
    def save_history(self):
        try:
            with open('data/books_data.json', 'w', encoding='utf-8') as f:
                json.dump(self.history, f, ensure_ascii=False, indent=2)
            messagebox.showinfo("Успех", "История успешно сохранена в data/books_data.json!")
        except Exception as e:
            messagebox.showerror("Ошибка", f"Не удалось сохранить историю: {str(e)}")
    
    def load_history(self):
        try:
            if os.path.exists('data/books_data.json'):
                with open('data/books_data.json', 'r', encoding='utf-8') as f:
                    self.history = json.load(f)
        except Exception as e:
            self.history = []
    
    def load_history_ui(self):
        self.load_history()
        self.update_history_display()
        messagebox.showinfo("Успех", "История успешно загружена!")

def main():
    root = tk.Tk()
    app = TaskGeneratorApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()
